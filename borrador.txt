def descargar_pdf(nro_boletin: int, path_archivo: str = None) -> str:
    """
    Descarga el PDF del boletín utilizando prioritariamente la ruta real (patharchivo)
    de la base de datos. Si no existe o falla, prueba los formatos históricos.
    """
    formatos_url = []

    # 1. PRIORIDAD ABSOLUTA: Si la BD tiene el patharchivo, construimos la URL con él
    if path_archivo:
        # Limpiamos espacios y barras iniciales por seguridad
        path_limpio = path_archivo.strip().lstrip('/')
        
        # Evaluamos cómo viene guardado en tu BD (ej: "boletines/2026/BO_16605.pdf" o solo "2026/BO_16605.pdf")
        if path_limpio.startswith("boletines/"):
            formatos_url.append(f"https://www.boletindigital.misiones.gov.ar/{path_limpio}")
        else:
            formatos_url.append(f"https://www.boletindigital.misiones.gov.ar/boletines/{path_limpio}")

    # Eliminamos URLs duplicadas manteniendo estrictamente el orden de prioridad
    formatos_url = list(dict.fromkeys(formatos_url))

    for url in formatos_url:
        print(f"Probando descarga desde URL: {url}")

        try:
            response = httpx.get(url, timeout=30, follow_redirects=True)

            # Si el servidor responde con 200 OK, encontramos el archivo correcto
            if response.status_code == 200:
                texto = ""
                with pdfplumber.open(io.BytesIO(response.content)) as pdf:
                    for page in pdf.pages:
                        texto += page.extract_text() or ""

                if texto.strip():
                    print(f"¡Éxito! Conectado y procesado desde la URL: {url}")
                    return texto.strip()

            print(f"No encontrado (Status {response.status_code}) en la ruta: {url}")

        except Exception as e:
            print(f"Error de red o procesamiento en {url}: {e}")
            continue

    print(f"Imposible descargar el boletín {nro_boletin}. Ningún patrón de URL coincidió.")
    return None
